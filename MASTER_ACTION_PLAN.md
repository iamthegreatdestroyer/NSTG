 * Meta-Testing: Use NSTG to test NSTG
 * - Generates tests for core packages

export class MetaTestGenerator {
  async generateTestsForNSTG() {
    // Use NSTG's own engine to test itself
    const corePackages = ['core', 'smt-solver', 'boundary-catalog'];

    for (const pkg of corePackages) {

      await this.validateTests(tests);
      await this.commitTests(tests);
    }
  }
